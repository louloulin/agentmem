47// Agent状态结构
48#[derive(Debug, Clone, Serialize, Deserialize)]
49pub struct AgentState {
50    pub id: String,
51    pub agent_id: u64,
52    pub session_id: u64,
53    pub timestamp: i64,
54    pub state_type: StateType,
55    pub data: Vec<u8>,
56    pub metadata: HashMap<String, String>,
57    pub version: u32,
58    pub checksum: u32,
59}
60
61impl AgentState {
62    pub fn new(
63        agent_id: u64,
64        session_id: u64,
65        state_type: StateType,
66        data: Vec<u8>,
67    ) -> Self {
68        let timestamp = chrono::Utc::now().timestamp();
69        let checksum = Self::calculate_checksum(&data);
70
71        Self {
72            id: Uuid::new_v4().to_string(),
73            agent_id,
74            session_id,
75            timestamp,
76            state_type,
77            data,
78            metadata: HashMap::new(),
79            version: 1,
80            checksum,
81        }
82    }
83
84    pub fn calculate_checksum(data: &[u8]) -> u32 {
85        data.iter().fold(0u32, |acc, &byte| acc.wrapping_add(byte as u32))
86    }
87
88    pub fn validate_checksum(&self) -> bool {
89        Self::calculate_checksum(&self.data) == self.checksum
90    }
91
92    pub fn update_data(&mut self, new_data: Vec<u8>) {
93        self.data = new_data;
94        self.checksum = Self::calculate_checksum(&self.data);
95        self.version += 1;
96        self.timestamp = chrono::Utc::now().timestamp();
97    }
98
99    pub fn set_metadata(&mut self, key: String, value: String) {
100        self.metadata.insert(key, value);
101    }
102
103    pub fn get_metadata(&self, key: &str) -> Option<&String> {
104        self.metadata.get(key)
105    }
106}

138// 记忆结构
139#[derive(Debug, Clone, Serialize, Deserialize)]
140pub struct Memory {
141    pub memory_id: String,
142    pub agent_id: u64,
143    pub memory_type: MemoryType,
144    pub content: String,
145    pub importance: f32,
146    pub embedding: Option<Vec<f32>>,
147    pub created_at: i64,
148    pub access_count: u32,
149    pub last_access: i64,
150    pub expires_at: Option<i64>,
151}
152
153impl Memory {
154    pub fn new(agent_id: u64, memory_type: MemoryType, content: String, importance: f32) -> Self {
155        let now = chrono::Utc::now().timestamp();
156        Self {
157            memory_id: Uuid::new_v4().to_string(),
158            agent_id,
159            memory_type,
160            content,
161            importance,
162            embedding: None,
163            created_at: now,
164            access_count: 0,
165            last_access: now,
166            expires_at: None,
167        }
168    }
169
170    pub fn access(&mut self) {
171        self.access_count += 1;
172        self.last_access = chrono::Utc::now().timestamp();
173    }
174
175    pub fn calculate_importance(&self, current_time: i64) -> f32 {
176        let time_decay = (current_time - self.created_at) as f32 / (24.0 * 3600.0);
177        let access_factor = (self.access_count as f32 + 1.0).ln();
178        self.importance * (-time_decay * 0.01).exp() * access_factor
179    }
180
181    pub fn is_expired(&self, current_time: i64) -> bool {
182        if let Some(expires_at) = self.expires_at {
183            current_time > expires_at
184        } else {
185            false
186        }
187    }
188}

190// 文档结构
191#[derive(Debug, Clone, Serialize, Deserialize)]
192pub struct Document {
193    pub id: String,  // 保持与 RAG 模块兼容
194    pub doc_id: String,
195    pub title: String,
196    pub content: String,
197    pub metadata: HashMap<String, String>,
198    pub chunks: Vec<DocumentChunk>,
199    pub created_at: i64,
200    pub updated_at: i64,
201}
202
203impl Document {
204    pub fn new(title: String, content: String) -> Self {
205        let now = chrono::Utc::now().timestamp();
206        let doc_id = Uuid::new_v4().to_string();
207        Self {
208            id: doc_id.clone(),  // 保持与 RAG 模块兼容
209            doc_id,
210            title,
211            content,
212            metadata: HashMap::new(),
213            chunks: Vec::new(),
214            created_at: now,
215            updated_at: now,
216        }
217    }
218
219    pub fn set_metadata(&mut self, key: String, value: String) {
220        self.metadata.insert(key, value);
221    }
222
223    pub fn chunk_document(&mut self, chunk_size: usize, overlap: usize) -> Result<(), AgentDbError> {
224        let mut chunks = Vec::new();
225        let content_bytes = self.content.as_bytes();
226        let mut pos = 0;
227        let mut chunk_index = 0;
228
229        while pos < content_bytes.len() {
230            let end = std::cmp::min(pos + chunk_size, content_bytes.len());
231            let chunk_content = String::from_utf8_lossy(&content_bytes[pos..end]).to_string();
232
233            let chunk_id = format!("{}_{}", self.doc_id, chunk_index);
234            let chunk = DocumentChunk {
235                id: chunk_id.clone(),  // 保持与 RAG 模块兼容
236                chunk_id,
237                doc_id: self.doc_id.clone(),
238                content: chunk_content,
239                chunk_index,
240                position: pos,
241                size: end - pos,
242                embedding: None,  // 初始时没有嵌入向量
243            };
244
245            chunks.push(chunk);
246            
247            if end >= content_bytes.len() {
248                break;
249            }
250            
251            pos = if chunk_size > overlap { 
252                pos + chunk_size - overlap 
253            } else { 
254                pos + 1 
255            };
256            chunk_index += 1;
257        }
258
259        self.chunks = chunks;
260        Ok(())
261    }
262}
